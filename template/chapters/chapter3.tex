\begin{algorithm}[H]
  \caption{示例算法（实际使用时替换为论文中的算法）}
  \label{alg:demo}
  \KwIn{数据集 $\mathcal{D}$，学习率 $\eta$，最大轮数 $T$}
  \KwOut{模型参数 $\theta$}
  初始化参数 $\theta_0$\;
  \For{$t = 1$ \KwTo $T$}{
    \For{每个批次 $(x, y) \in \mathcal{D}$}{
      计算损失 $\mathcal{L}(\theta; x, y)$\;
      计算梯度 $g \leftarrow \nabla_\theta \mathcal{L}$\;
      更新参数 $\theta \leftarrow \theta - \eta \cdot g$\;
    }
    \If{验证集损失不再下降}{
      \Break\;
    }
  }
  \Return $\theta$\;
\end{algorithm}
